# Ingress install
Create shared ELB, note down EIP and EIP id

kubectl create ns ingress-nginx

helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
helm install ingress-nginx ingress-nginx/ingress-nginx --version 4.8.2


helm install ingress-nginx ingress-nginx/ingress-nginx \
--set controller.service.loadBalancerIP="80.158.108.47" \
--set controller.service.annotations."kubernetes\.io/elb\.class"=union \
--set controller.service.annotations."kubernetes\.io/elb\.id"=86147271-d8cf-4efd-8b87-342abe1b5464 \
-n ingress-nginx
